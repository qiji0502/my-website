# 产品数据分析工具集

import pandas as pd
import numpy as np
from scipy import stats
import json

# ========================================
# 1. A/B测试分析函数
# 使用场景：评估产品改版、功能优化的效果
# 输入：两组实验数据（访问量、转化数）
# 输出：统计显著性结论与建议
# ========================================
def ab_test_analysis(visitors_a, clicks_a, visitors_b, clicks_b, confidence_level=0.95):
    """
    A/B测试统计分析
    参数:
        visitors_a, clicks_a: 方案A的访问量和点击数
        visitors_b, clicks_b: 方案B的访问量和点击数
        confidence_level: 置信水平，默认95%
    返回:
        dict: 包含转化率、提升、显著性、建议    """
    
    # 计算转化率
    cr_a = clicks_a / visitors_a
    cr_b = clicks_b / visitors_b
    
    # 计算提升率
    uplift = (cr_b - cr_a) / cr_a * 100
    
    # Z检验计算统计显著性
    se_a = np.sqrt(cr_a * (1 - cr_a) / visitors_a)
    se_b = np.sqrt(cr_b * (1 - cr_b) / visitors_b)
    z_score = (cr_b - cr_a) / np.sqrt(se_a**2 + se_b**2)
    
    # 判断临界值
    z_critical = 1.96 if confidence_level == 0.95 else 2.58
    
    # 结果解读
    is_significant = abs(z_score) > z_critical
    
    result = {
        '方案A转化率': f"{cr_a:.2%}",
        '方案B转化率': f"{cr_b:.2%}",
        '相对提升': f"{uplift:.1f}%",
        '统计显著性': '✅显著' if is_significant else '❌ 不显著',
        '建议': '推荐上线方案B' if is_significant and uplift > 0 else '保留方案A或继续测试',
        '样本量': visitors_a + visitors_b,
        '置信水平': f"{confidence_level*100}%"
    }
    
    return result

# 2. RFM用户分群模型
# 使用场景：用户精细化运营、个性化推荐
# 输入：用户最近购买时间、购买频率、消费金额
# 输出：用户等级与运营策略
# ========================================
def rfm_segmentation(recency, frequency, monetary):
    """
    RFM用户分群
    参数:
        recency: 最近购买天数
        frequency: 购买频率（次/月）
        monetary: 消费金额（元）
    返回:
        tuple: (用户等级, 运营策略, 详细评分)
    """
    
    # 评分标准（可调整阈值）
    r_score = 3 if recency <= 30 else (2 if recency <= 60 else 1)
    f_score = 3 if frequency >= 3 else (2 if frequency >= 1 else 1)
    m_score = 3 if monetary >= 500 else (2 if monetary >= 200 else 1)
    
    total = r_score + f_score + m_score
    
    # 用户分层
    if total >= 8:
        segment = "高价值用户"
        strategy = "专属权益 · 新品优先 · VIP服务 · 深度调研"
 elif total >= 5:
        segment = "潜力用户"
        strategy = "个性化推荐 · 激励复购 · 优惠券 · 内容营销"
    else:
        segment = "一般用户"
        strategy = "唤醒策略 · 基础服务 · 降低流失 · 普惠活动"
    
    return segment, strategy, {'R': r_score, 'F': f_score, 'M': m_score, '总分': total}

# ========================================
# 3. 功能优先级评估（RICE模型）
# 使用场景：产品需求排期、资源分配
# 输入：覆盖用户数、影响力、信心指数、开发成本
# 输出：优先级分数与建议
# ========================================
def rice_prioritization(reach, impact, confidence, effort):
    """
    RICE优先级评估
    参数:
        reach: 覆盖用户数（万）
        impact: 影响力（1-3分）
        confidence: 信心指数（0-100%）
        effort: 开发成本（人天）
    返回:
        dict: 优先级分数、等级、建议
    """
    
    # RICE公式
    rice_score = (reach * impact * confidence) / effort
    
    # 优先级分级
    if rice_score > 100:
        priority = "🔥 高优先级"
        recommendation = "立即启动，安排开发"
    elif rice_score > 50:
        priority = "⚖️ 中优先级"
        recommendation = "规划排期，Q4实施"
    else:
        priority = "📋 低优先级"
        recommendation = "放入需求池，持续观察"
    
    # ROI预估
    roi = (reach * impact * confidence / effort) / 10
    
    return {
        'RICE分数': f"{rice_score:.1f}",
        '优先级': priority,
        '建议': recommendation,
        'ROI预估': f"{roi:.1f}倍",
        '覆盖用户': f"{reach}万",
        '开发成本': f"{effort}人天"
    }

# 4. 数据自动分析与洞察生成
# 使用场景：上传数据后自动生成产品报告
# 输入：DataFrame格式的用户行为数据
# 输出：关键指标、异常检测、优化建议
# ========================================
def auto_analysis(data):
    """
    自动数据分析与洞察生成
    参数:
        data: pandas DataFrame，包含用户行为数据
    返回:
        dict: 分析报告
    """
    
    report = {}
    
    # 基础统计
    report['基础统计'] = {
        '总记录数': len(data),
        '唯一用户数': data['user_id'].nunique(),
        '数据时间范围': f"{data['timestamp'].min()} 至 {data['timestamp'].max()}"
    }
    
    # 行为分析
    if 'behavior' in data.columns:
        behavior_dist = data['behavior'].value_counts()
        report['行为分布'] = behavior_dist.to_dict()
 report['热门行为'] = behavior_dist.index[0]
    
    # 价值分析
    if 'amount' in data.columns:
        avg_amount = data['amount'].mean()
        report['平均消费'] = f"¥{avg_amount:.2f}"
        # 高价值用户识别
        high_value = data[data['amount'] > avg_amount * 2]
        report['高价值用户数'] = len(high_value)
 report['高价值用户占比'] = f"{len(high_value)/len(data)*100:.1f}%"
    
    # 满意度分析
    if 'satisfaction' in data.columns:
        avg_satisfaction = data['satisfaction'].mean()
        report['平均满意度'] = f"{avg_satisfaction:.2f}/5"
        
        # 满意度预警
        low_satisfaction = data[data['satisfaction'] < 3]
        report['低满意度用户'] = len(low_satisfaction)
        report['满意度预警'] = "需要关注" if len(low_satisfaction) > 0 else "良好"
    
    # 异常检测
    if 'amount' in data.columns:
        Q1 = data['amount'].quantile(0.25)
        Q3 = data['amount'].quantile(0.75)
        IQR = Q3 - Q1
        outliers = data[(data['amount'] < Q1 - 1.5 * IQR) | (data['amount'] > Q3 + 1.5 * IQR)]
        report['异常交易数'] = len(outliers)
        report['异常检测'] = f"检测到 {len(outliers)} 个异常值" if len(outliers) > 0 else "无异常"
    
    # 产品优化建议
    suggestions = []
    if report.get('满意度预警') == "需要关注":
        suggestions.append("1. 针对低满意度用户进行回访，识别核心痛点")
    if report.get('高价值用户占比') and float(report['高价值用户占比'].strip('%')) < 10:
        suggestions.append("2. 高价值用户占比偏低，建议推出VIP权益提升留存")
    if report.get('异常交易数', 0) > 0:
        suggestions.append("3. 存在异常交易，建议进行风控核查")
    if len(suggestions) == 0:
        suggestions.append("1. 数据表现健康，建议继续保持当前策略")
        suggestions.append("2. 关注核心指标波动，定期复盘")
    
    report['优化建议'] = suggestions
    
    return report
